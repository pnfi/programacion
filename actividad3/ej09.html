Uso del if(){}
Se usa para verificar el contenido de una variable, es una condicional que significa "si se cumple esto hacer aquello"

<script type="text/javascript">
    var hora = prompt('Me puedes decir la Hora de la 0 a la 23 ?');
    if(hora >= 0 && hora <= 5 ){document.write('Felíz Madrugada <br>')}
    if(hora >= 6 && hora <= 11 ){document.write('Buen día <br>')}
    if(hora >= 12 && hora <= 18 ){document.write('Buenas Tardes <br>')}
    if(hora >= 19 && hora <= 23 ){document.write('Buenas Noches <br>')}
    if(hora > 23 ){document.write('Pusiste algo mal en tu respuesta... de la hora <br>')}

    var dia = prompt('Dame un Numero del 1 al 7 y te daré un día de la semana');
    if(dia==1){document.write ('El número '+ dia + ' corresponde al día Domingo')}
    if(dia==2){document.write ('El número '+ dia + ' corresponde al día Lunes')}
    if(dia==3){document.write ('El número '+ dia + ' corresponde al día Martes')}
    if(dia==4){document.write ('El número '+ dia + ' corresponde al día Miercoles')}
    if(dia==5){document.write ('El número '+ dia + ' corresponde al día Jueves')}
    if(dia==6){document.write ('El número '+ dia + ' corresponde al día Viernes')}
    if(dia==7){document.write ('El número '+ dia + ' corresponde al día Sabado')}
    if(dia>7){document.write ('Pusiste algo mal en tu respuesta... del día')}
