Ejemplo 06 - Capturar y sumar 2 variables con prompt
Sumar y multiplicar, con variables proporcionadas por el usuario
Ejemplo:
Código: 

<script type="text/javascript">
	alert("Te pediré 2 numeros para sumarlos, espero me des una buena cantidad.");
	var num1 = parseInt(prompt("Tipea el Primer Numero por favor"));
	var num2 = parseInt(prompt("Tipea el Segundo Numero por favor"));
	document.write("El primer numero es: "+num1+"</br>"+
					"El segundo numero es: "+num2+"</br>"+
					"La suma de ambos es: "+(num1+num2)+"</br>"+
					"El 10 % es: "+((num1+num2)*.1)+"</br>"+
					"La suma de ambos + el 10% es: "+((num1+num2)*1.1)
					);
</script>
